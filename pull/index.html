#! /bin/sh # 

banner()
{
  echo "#################################################"
  echo "# $@"
  echo "#################################################"
}

#
# PULL
#

for f in  \
 cloudmesh-common \
 cloudmesh-cmd5 \
 cloudmesh-sys \
 cloudmesh-cloud \
 cloudmesh-storage \
 cloudmesh-emr \
 cloudmesh-openapi \
 cloudmesh-nn \
 cloudmesh-nist \
 cloudmesh-inventory \
 cloudmesh-conda \
 get \
 about \
 cloudmesh.github.io \
 cloudmesh-community.github.io
do
    banner pull $f
    cd $f
    git pull
    cd ..
done

for f in cloudmesh-common cloudmesh-cmd5 cloudmesh-sys cloudmesh-cloud
do
    banner pip install $f
    cd $f
    python setup.py install
done

cd ..

exit 0
