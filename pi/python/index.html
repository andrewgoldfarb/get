#! /bin/sh # # curl -Ls http://cloudmesh.github.io/get/pi/python | sh # # To specify a specific version: # # VERSION=3.8.2 curl -Ls http://cloudmesh.github.io/get/pi/python | sh # # The default VERSION is 3.8.2 # # 

start=`date +%s`

if [ -z "$VERSION" ]; then
    VERSION='3.8.2'
fi
PYTHON="python${VERSION%.*}"
PIP="pip${VERSION%.*}"

NORMAL='\033[0m'
RED='\033[0;31m'

AddToBashrc(){
grep -qF -- "${1}" ~/.bashrc || echo "${1}" >> "${HOME}/.bashrc"
}

Echo() {
 echo "${RED}${1}${NORMAL}"
}

Echo "#################################################"
Echo "# Raspbery Pi Python ${VERSION} Installation"
Echo "#################################################"
echo
echo "Aproximate instalation time: 11 minutes"
echo
Echo "# -----------------------------------------------"
Echo "# Update the system"
Echo "# -----------------------------------------------"
startupdate=`date +%s`

sudo apt-get update
sudo apt-get full-upgrade
sudo apt-get install -y build-essential \
    tk-dev\
    libncurses5-dev\
    libncursesw5-dev\
    libreadline6-dev\
    libdb5.3-dev\
    libgdbm-dev\
    libsqlite3-dev\
    libssl-dev\
    libbz2-dev\
    libexpat1-dev\
    liblzma-dev\
    zlib1g-dev\
    libffi-dev
endupdate=`date +%s`
echo
Echo "# -----------------------------------------------"
Echo "# Download the source"
Echo "# -----------------------------------------------"
startdownload=`date +%s`
wget https://www.python.org/ftp/python/${VERSION}/Python-${VERSION}.tar.xz
enddownload=`date +%s`
echo
Echo "# -----------------------------------------------"
Echo "# Compile Python"
Echo "# -----------------------------------------------"
startpython=`date +%s`
tar xf Python-${VERSION}.tar.xz
cd Python-${VERSION}
./configure --prefix=/usr/local/opt/python-${VERSION}
make -j echo
Echo "# -----------------------------------------------"
Echo "# Install Python"
Echo "# -----------------------------------------------"
sudo make altinstall
AddToBashrc "export PATH=$PATH:/usr/local/opt/python-${VERSION}/bin"
. ~/.bashrc

python -m pip install -U pip
python -m pip install -U matplotlib

endpython=`date +%s`

Echo "#################################################"
Echo "# Install of Python ${VERSION} Completed"
Echo "#################################################"
end=`date +%s`
runtime=$((end-start))
downloadtime=$((end-start))
pythontime=$((end-start))
updatetime=$((end-start))

echo

echo "Time to update and upgarde: ${updatetime} s"
echo "Time to download Python:    ${downloadtime} s"
echo "Time to install Python:     ${pythontime} s"

echo "Time for total install:     ${runtime} s"
echo
Echo "#################################################"

exit 0
