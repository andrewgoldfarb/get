#! /bin/sh # curl -Ls http://cloudmesh.github.io/get/pi | sh # 

NORMAL='\033[0m'
RED='\033[0;31m'

Echo() {
 echo "${RED}${1}${NORMAL}"
}

Echo "#################################################"
Echo "# Raspbery Pi Cloudmesh Cluster Instalation     #"
Echo "#################################################"
echo
Echo "#"
Echo "# Update the system"
Echo "#"
sudo apt-get update -y
sudo apt-get full-upgrade -y
echo
Echo "#"
Echo "# Create a Python Virtual Env ~/ENV3"
Echo "#"
python3 -m venv ~/ENV3
. ~/ENV3/bin/activate
pip install pip -U
which python
which pip
python3 --version
pip --version
Echo "#"
Echo "# Install cloudmesh-installer"
Echo "#"
pip install cloudmesh-installer -U
Echo "#"
Echo "# Add ENV3 to bashrc"
Echo "#"
LINE="source ${HOME}/ENV3/bin/activate"
grep -qF -- "$LINE" ~/.bashrc || echo "$LINE" >> ~/.bashrc
echo
Echo "Please activate with"
echo
Echo "    source ~/ENV3/bin/activate"
echo
exit 0
