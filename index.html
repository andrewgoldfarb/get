#! /bin/sh # 

banner()
{
  echo "#################################################"
  echo "# clone -> $@"
  echo "#################################################"
}

cloudmesh_clone()
{
  echo "#################################################"
  echo "# clone -> $1"
  echo "#################################################"
  git clone https://github.com/cloudmesh/$1.git

}

community_clone()
{
  echo "#################################################"
  echo "# clone -> $1"
  echo "#################################################"
  git clone https://github.com/cloudmesh-community/$1.git

}

#
# CLONE
#
cloudmesh_clone  cloudmesh-common
cloudmesh_clone  cloudmesh-cmd5
cloudmesh_clone  cloudmesh-sys
cloudmesh_clone  cloudmesh-cloud
cloudmesh_clone  cloudmesh-storage
cloudmesh_clone  cloudmesh-emr
cloudmesh_clone  cloudmesh-comet
cloudmesh_clone  cloudmesh-openapi
cloudmesh_clone  cloudmesh-nn
cloudmesh_clone  cloudmesh-nist
cloudmesh_clone  cloudmesh-conda
cloudmesh_clone  cloudmesh-cloudmeshhub.io
cloudmesh_clone  cloudmesh-get
cloudmesh_clone  cloudmesh-about
community_clone cloudmesh-community.github.io

for f in cloudmesh-common cloudmesh-cmd5 cloudmesh-sys cloudmesh-cloud
do
  banner install $f
  cd $f
  pip install -r requirements.txt
  # python setup.py install
  pip install .
  cd ..
done

exit 0
