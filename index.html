#! /bin/sh # 

banner()
{
  echo "#################################################"
  echo "# clone -> $@"
  echo "#################################################"
}

cloudmesh-clone()
{
  echo "#################################################"
  echo "# clone -> $1"
  echo "#################################################"
  git clone https://github.com/cloudmesh/$1.git

}

community-clone()
{
  echo "#################################################"
  echo "# clone -> $1"
  echo "#################################################"
  git clone https://github.com/cloudmesh-community/$1.git

}

#
# CLONE
#
cloudmesh-clone  cloudmesh-common
cloudmesh-clone  cloudmesh-cmd5
cloudmesh-clone  cloudmesh-sys
cloudmesh-clone  cloudmesh-cloud
cloudmesh-clone  cloudmesh-storage
cloudmesh-clone  cloudmesh-emr
cloudmesh-clone  cloudmesh-comet
cloudmesh-clone  cloudmesh-openapi
cloudmesh-clone  cloudmesh-nn
cloudmesh-clone  cloudmesh-nist
cloudmesh-clone  cloudmesh-conda
cloudmesh-clone  cloudmesh-cloudmeshhub.io
cloudmesh-clone  cloudmesh-get
cloudmesh-clone  cloudmesh-about
community-clone cloudmesh-community.github.io

for f in cloudmesh-common cloudmesh-cmd5 cloudmesh-sys cloudmesh-cloud
do
  banner install $f
  cd $f
  python setup.py install
  pip install .
  cd ..
done

exit 0
