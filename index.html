#! /bin/sh # 


banner()
{
  echo "#################################################"
  echo "# $@"
  echo "#################################################"
}


#
# CORE
#
for f in common cmd5 sys cloud storage emr comet
do
   banner clone $f

   git clone https://github.com/cloudmesh/cloudmesh-$f.git
done

#
# OPEN API
#

for f in openapi nn nist
do
   banner clone $f
   git clone https://github.com/cloudmesh/cloudmesh-$f.git
done

#
# EXPERIMENTAL
#
for f in inventory conda
do
   banner clone $f
   git clone https://github.com/cloudmesh/cloudmesh-$f.git
done

#
# MANAGEMENT ABD WEB PAGE
#

for f in git about  cloudmesh.github.io
do
   banner clone $f
   git clone https://github.com/cloudmesh/$f.git
done

for f in cloudmesh-community.github.io
do
   banner clone $f
   git clone https://github.com/cloudmesh/$f.git
done

for f in cloudmesh-common cloudmesh-cmd5 cloudmesh-sys cloudmesh-cloud
do
  banner install $f
  cd $f
  python setup.py install
  pip install .
  cd ..
done


exit 0
